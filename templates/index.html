import os import zipfile import tempfile import numpy as np import pandas as pd import geopandas as gpd import matplotlib matplotlib.use("Agg") import matplotlib.pyplot as plt from flask import Flask, render_template, request, jsonify, send_file from shapely.geometry import Polygon, Point, LineString, box, MultiPolygon app = Flask(__name__) UPLOAD = "uploads" OUTPUT = "outputs" STATIC = "static" os.makedirs(UPLOAD, exist_ok=True) os.makedirs(OUTPUT, exist_ok=True) os.makedirs(STATIC, exist_ok=True) # ========================= # CRS # ========================= def get_crs(zone): return "EPSG:32644" if str(zone) == "44" else "EPSG:32645" # ========================= # SAFE GEOMETRY # ========================= def normalize_geom(geom): if geom is None: return [] if isinstance(geom, Polygon): return [geom] if isinstance(geom, MultiPolygon): return [g for g in geom.geoms if g.is_valid] return [] # ========================= # SHAPEFILE ZIP LOAD # ========================= def load_shapefile(zip_path): tmp = tempfile.mkdtemp() with zipfile.ZipFile(zip_path, "r") as z: z.extractall(tmp) for f in os.listdir(tmp): if f.endswith(".shp"): gdf = gpd.read_file(os.path.join(tmp, f)) return gdf.geometry.unary_union raise Exception("No shapefile found") # ========================= # POINTS # ========================= def build_points(df): df = df.copy() df["X"] = pd.to_numeric(df["X"], errors="coerce") df["Y"] = pd.to_numeric(df["Y"], errors="coerce") df = df.dropna(subset=["X", "Y"]) return df, [Point(xy) for xy in zip(df["X"], df["Y"])] # ========================= # WHOLE BOUNDARY # ========================= def build_whole(df, order_mode="manual"): df, pts = build_points(df) if order_mode == "auto": df = df.sort_values(["X", "Y"]) else: df = df.sort_values("Order") coords = list(zip(df["X"], df["Y"])) if len(coords) < 3: raise Exception("Not enough points") if coords[0] != coords[-1]: coords.append(coords[0]) poly = Polygon(coords) if not poly.is_valid: poly = poly.buffer(0) line = LineString(coords) return poly, line, pts # ========================= # SEGMENTED # ========================= def build_segmented(df): df, _ = build_points(df) if "Compartment" not in df.columns: raise Exception("Compartment missing") result = [] for comp, g in df.groupby("Compartment"): g = g.sort_values("Order") coords = list(zip(g["X"], g["Y"])) if len(coords) < 3: continue if coords[0] != coords[-1]: coords.append(coords[0]) poly = Polygon(coords) if not poly.is_valid: poly = poly.buffer(0) line = LineString(coords) pts = [Point(xy) for xy in coords] result.append((comp, poly, line, pts)) return result # ========================= # SAMPLE FISHNET (FIXED FULL) # ========================= def build_sample(poly, rows, cols, cell_w, cell_h): minx, miny, maxx, maxy = poly.bounds dx = (maxx - minx) / cols dy = (maxy - miny) / rows dx = cell_w if cell_w else dx dy = cell_h if cell_h else dy pts = [] for i in range(cols): for j in range(rows): x1 = minx + i * dx y1 = miny + j * dy cell = box(x1, y1, x1 + dx, y1 + dy) if cell.intersects(poly): pts.append(cell.centroid) return pts # ========================= # PREVIEW # ========================= def save_plot(): path = os.path.join(STATIC, "preview.png") plt.savefig(path, dpi=150, bbox_inches="tight") plt.close() return "/static/preview.png" def preview_whole(poly, line, pts): fig, ax = plt.subplots() x, y = poly.exterior.xy ax.plot(x, y, "g-") x, y = line.xy ax.plot(x, y, "k-") for p in pts: ax.scatter(p.x, p.y, s=10, c="blue") ax.set_axis_off() return save_plot() def preview_segmented(seg): fig, ax = plt.subplots() for comp, poly, line, pts in seg: x, y = poly.exterior.xy ax.plot(x, y) x, y = line.xy ax.plot(x, y, "k-") for p in pts: ax.scatter(p.x, p.y, s=8) c = poly.centroid ax.text(c.x, c.y, str(comp)) ax.set_axis_off() return save_plot() def preview_sample(poly, pts): fig, ax = plt.subplots() x, y = poly.exterior.xy ax.plot(x, y, "r-") for p in pts: ax.scatter(p.x, p.y, s=10) ax.set_axis_off() return save_plot() # ========================= # ROUTES # ========================= @app.route("/") def home(): return render_template("index.html") @app.route("/preview", methods=["POST"]) def preview(): file = request.files["file"] mode = request.form.get("mode") zone = request.form.get("zone") path = os.path.join(UPLOAD, file.filename) file.save(path) try: # ZIP if path.endswith(".zip"): geom = normalize_geom(load_shapefile(path)) poly = geom[0] return jsonify({"image": preview_whole( poly, LineString(poly.exterior.coords), [Point(xy) for xy in poly.exterior.coords] )}) # EXCEL df = pd.read_excel(path) if mode == "boundary": poly, line, pts = build_whole(df, request.form.get("order_mode")) return jsonify({"image": preview_whole(poly, line, pts)}) if mode == "compartment": seg = build_segmented(df) return jsonify({"image": preview_segmented(seg)}) if mode == "sample": poly, _, _ = build_whole(df, "manual") rows = int(request.form.get("rows", 10)) cols = int(request.form.get("cols", 10)) cw = float(request.form.get("cell_width", 0)) ch = float(request.form.get("cell_height", 0)) pts = build_sample(poly, rows, cols, cw, ch) return jsonify({"image": preview_sample(poly, pts)}) return jsonify({"error": "Invalid mode"}) except Exception as e: return jsonify({"error": str(e)}) # ========================= if __name__ == "__main__": app.run(debug=True)
