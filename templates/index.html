import os import zipfile import pandas as pd import geopandas as gpd import streamlit as st import matplotlib.pyplot as plt from shapely.geometry import Point, LineString, Polygon from io import BytesIO # ========================= # PAGE CONFIG # ========================= st.set_page_config( page_title="Community Forest GIS Tool", page_icon="🌲", layout="wide" ) OUTPUT_FOLDER = "OUTPUT" os.makedirs(OUTPUT_FOLDER, exist_ok=True) # ========================= # HEADER UI # ========================= st.title("🌲 Community Forest GIS Processing System") st.caption("Generate Forest Boundaries & Compartment Maps (Shapefile + ZIP Export)") # ========================= # SIDEBAR UI # ========================= with st.sidebar: st.header("⚙️ Settings") mode = st.selectbox( "Processing Mode", ["Whole Forest Boundary", "Segmented Forest (Compartment-wise)"] ) zone = st.radio("UTM Zone", ["44", "45"], index=1) st.divider() st.info("📌 Upload Excel file with required columns before running processing.") # ========================= # FILE UPLOAD UI # ========================= st.subheader("📤 Input Data") uploaded_file = st.file_uploader( "Upload Excel File", type=["xlsx", "xls"], help="Required columns depend on selected mode" ) # Dynamic column hint if mode == "Whole Forest Boundary": st.markdown("**Required Columns:** `Forest, X, Y, Order`") else: st.markdown("**Required Columns:** `Forest, Compartment, X, Y, Order`") # ========================= # HELP FUNCTIONS # ========================= def get_crs(zone): return "EPSG:32644" if zone == "44" else "EPSG:32645" def make_zip(folder_path): zip_buffer = BytesIO() with zipfile.ZipFile(zip_buffer, "w", zipfile.ZIP_DEFLATED) as zipf: for root, _, files in os.walk(folder_path): for file in files: file_path = os.path.join(root, file) arcname = os.path.relpath(file_path, folder_path) zipf.write(file_path, arcname) zip_buffer.seek(0) return zip_buffer # ========================= # RUN BUTTON UI # ========================= run = st.button("🚀 Run GIS Processing", use_container_width=True) # ========================= # PROCESS FUNCTION (SAME LOGIC SHORTENED UI STYLE) # ========================= def process_whole(file, crs): df = pd.read_excel(file) grouped = df.groupby("Forest") results = {} progress = st.progress(0) for i, (forest, group) in enumerate(grouped): group = group.sort_values("Order") group["X"] = pd.to_numeric(group["X"], errors="coerce") group["Y"] = pd.to_numeric(group["Y"], errors="coerce") group = group.dropna() coords = list(zip(group["X"], group["Y"])) if len(coords) < 3: continue if coords[0] != coords[-1]: coords.append(coords[0]) polygon = Polygon(coords).buffer(0) if not polygon.is_valid: continue line = LineString(coords) folder = os.path.join(OUTPUT_FOLDER, f"whole_{forest}") os.makedirs(folder, exist_ok=True) gpd.GeoDataFrame(group.copy(), geometry=[Point(xy) for xy in coords[:len(group)]], crs=crs)\ .to_file(os.path.join(folder, "points.shp")) gpd.GeoDataFrame(geometry=[line], crs=crs)\ .to_file(os.path.join(folder, "line.shp")) gpd.GeoDataFrame({"Forest": [forest]}, geometry=[polygon], crs=crs)\ .to_file(os.path.join(folder, "polygon.shp")) results[forest] = make_zip(folder) progress.progress((i + 1) / len(grouped)) return results def process_segmented(file, crs): df = pd.read_excel(file) grouped = df.groupby("Forest") results = {} progress = st.progress(0) total = len(grouped) for i, (forest, group) in enumerate(grouped): all_polys = [] all_lines = [] all_points = [] group["X"] = pd.to_numeric(group["X"], errors="coerce") group["Y"] = pd.to_numeric(group["Y"], errors="coerce") group = group.dropna() for comp, g in group.groupby("Compartment"): g = g.sort_values("Order") coords = list(zip(g["X"], g["Y"])) if len(coords) < 3: continue if coords[0] != coords[-1]: coords.append(coords[0]) poly = Polygon(coords).buffer(0) if not poly.is_valid: continue line = LineString(coords) all_polys.append({"Comp": comp, "geometry": poly}) all_lines.append({"Comp": comp, "geometry": line}) for _, r in g.iterrows(): all_points.append({"geometry": Point(r["X"], r["Y"])}) folder = os.path.join(OUTPUT_FOLDER, f"seg_{forest}") os.makedirs(folder, exist_ok=True) gpd.GeoDataFrame(all_points, crs=crs)\ .to_file(os.path.join(folder, "points.shp")) gpd.GeoDataFrame(all_lines, crs=crs)\ .to_file(os.path.join(folder, "lines.shp")) gpd.GeoDataFrame(all_polys, crs=crs)\ .to_file(os.path.join(folder, "polygons.shp")) # Preview fig, ax = plt.subplots(figsize=(5, 5)) gpd.GeoDataFrame(all_polys, crs=crs).plot(ax=ax, alpha=0.4) gpd.GeoDataFrame(all_points, crs=crs).plot(ax=ax, color="red", markersize=10) ax.set_title(forest) ax.axis("off") st.pyplot(fig) results[forest] = make_zip(folder) progress.progress((i + 1) / total) return results # ========================= # EXECUTION UI FLOW # ========================= if run: if uploaded_file is None: st.warning("⚠️ Please upload an Excel file first.") st.stop() st.markdown("---") st.subheader("📊 Processing Status") crs = get_crs(zone) with st.spinner("Processing GIS data... please wait"): if mode == "Whole Forest Boundary": results = process_whole(uploaded_file, crs) else: results = process_segmented(uploaded_file, crs) if not results: st.error("No valid outputs generated.") else: st.success("✅ Processing Completed Successfully!") st.subheader("📦 Download Outputs") cols = st.columns(2) for i, (name, zip_buffer) in enumerate(results.items()): with cols[i % 2]: st.download_button( label=f"⬇ Download {name}", data=zip_buffer, file_name=f"{name}.zip", use_container_width=True )
